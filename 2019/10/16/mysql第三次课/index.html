CREATE DATABASE SchoolDB1;
USE SchoolDB1;
CREATE TABLE student(
sid INT PRIMARY KEY AUTO_INCREMENT,
snam VARCHAR(20) NOT NULL,
sex VARCHAR(5),
sage INT,
stel VARCHAR(18) UNIQUE
)
CREATE TABLE Course(
cid INT PRIMARY KEY AUTO_INCREMENT,
cnam VARCHAR(50) NOT NULL
)
CREATE TABLE Score(
scid INT PRIMARY KEY AUTO_INCREMENT,
sid INT,
cid INT,
score INT
)

#查询学生信息，并按年纪进行排序
#asc 升序 desc 降序
SELECT * FROM student ORDER BY sage ASC;
#查询出每个学生的平均成绩
SELECT sid,AVG(score) ‘平均成绩’
FROM score GROUP BY sid;
#查询出每个学生的平均成绩，要显示的字段
#为学生姓名、平均成绩。
#普通连接
#连接之后需要加连接条件，把正确的数据
#给筛选出来
SELECT snam,AVG(score) ‘平均成绩’ FROM
student s,score c
WHERE
s.sid=c.sid GROUP BY snam;
#内连接
#inner join….on
SELECT snam,AVG(score) ‘平均成绩’ FROM
student s INNER JOIN score c
ON
s.sid=c.sid
GROUP BY snam;
#查询出没有三门课成绩的学生信息
SELECT * FROM student WHERE sid IN(
SELECT sid FROM score GROUP BY sid
HAVING COUNT()<3);
